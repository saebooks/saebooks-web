Keys with empty/None values are dropped. ``target`` is the HTMX swap target — if omitted, links degrade gracefully to plain anchors. #} {%- macro _qs(filters) -%} {%- for k, v in (filters or {}).items() if v -%} &{{ k }}={{ v }} {%- endfor -%} {%- endmacro -%} {%- macro _page_url(base_path, page_offset, limit, filters, fragment) -%} {{ base_path }}?offset={{ page_offset }}&limit={{ limit }}{{ _qs(filters) }}{% if fragment %}#{{ fragment }}{% endif %} {%- endmacro -%} {% macro _page_btn(label, page_offset, base_path, limit, filters, target, active=False, disabled=False, label_class='', fragment=None) -%} {% set url = _page_url(base_path, page_offset, limit, filters, fragment) %} {% if disabled %} {{ label | safe }} {% elif active %} {{ label | safe }} {% else %} {{ label | safe }} {% endif %} {%- endmacro %} {# Build the list of page numbers to show. With ≤ 7 pages, show all. With more, always show first + last + current ± 2, with ellipses between gaps. Returns a list where each entry is either an int (page number) or the string '…' (ellipsis spacer). #} {%- macro _page_window(current, total_pages) -%} {%- if total_pages <= 7 -%} {%- for p in range(1, total_pages + 1) -%}{{ p }}{% if not loop.last %},{% endif %}{%- endfor -%} {%- else -%} {%- set ns = namespace(out=[]) -%} {%- set _ = ns.out.append(1) -%} {%- if current - 2 > 2 -%}{%- set _ = ns.out.append('…') -%}{%- endif -%} {%- for p in range([current - 2, 2]|max, [current + 2, total_pages - 1]|min + 1) -%} {%- set _ = ns.out.append(p) -%} {%- endfor -%} {%- if current + 2 < total_pages - 1 -%}{%- set _ = ns.out.append('…') -%}{%- endif -%} {%- set _ = ns.out.append(total_pages) -%} {%- for p in ns.out -%}{{ p }}{% if not loop.last %},{% endif %}{%- endfor -%} {%- endif -%} {%- endmacro -%} {% macro page_numbers(base_path, total, limit, offset, filters=None, target=None, fragment=None) -%} {% if total and limit and total > 0 %} {% set total_pages = ((total + limit - 1) // limit) %} {% set current_page = (offset // limit) + 1 if limit > 0 else 1 %} {% set prev_offset = (current_page - 2) * limit if current_page > 1 else None %} {% set next_offset = current_page * limit if current_page < total_pages else None %} {% set window = _page_window(current_page, total_pages) | trim %}
